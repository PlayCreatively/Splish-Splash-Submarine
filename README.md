# GlobalSettings
- PLAYER
	-- HP

	-- HP

- SCREEN
	-- visible ratio (20/80)
- RADAR
	-- scan frequency
	-- blip duration
- BULLET
	-- bullet speed
	-- shoot frequency
	-- reload speed
